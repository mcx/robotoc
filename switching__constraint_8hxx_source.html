28inline void SwitchingConstraint::evalSwitchingConstraint(
29 Robot& robot, const ImpulseStatus& impulse_status, const double dt1,
30 const double dt2, const SplitSolution& s,
31 SwitchingConstraintResidual& sc_residual) {
32 assert(dt1 > 0);
33 assert(dt2 > 0);
34 sc_residual.setImpulseStatus(impulse_status);

36 robot.integrateConfiguration(s.q, dq_, 1.0, q_);

44inline void SwitchingConstraint::linearizeSwitchingConstraint(
45 Robot& robot, const ImpulseStatus& impulse_status, const double dt1,
46 const double dt2, const SplitSolution& s, SplitKKTMatrix& kkt_matrix,
47 SplitKKTResidual& kkt_residual, SwitchingConstraintJacobian& sc_jacobian,
48 SwitchingConstraintResidual& sc_residual) {
49 assert(dt1 > 0);
50 assert(dt2 > 0);
51 sc_residual.setImpulseStatus(impulse_status);
52 sc_jacobian.setImpulseStatus(impulse_status);
53 sc_residual.setZero();
54 sc_jacobian.setZero();
55 evalSwitchingConstraint(robot, impulse_status, dt1, dt2, s, sc_residual);
56 robot.computeContactPositionDerivative(impulse_status, sc_jacobian.Pq());
57 if (has_floating_base_) {
58 robot.dIntegrateTransport_dq(s.q, dq_, sc_jacobian.Pq(), sc_jacobian.Phiq());
59 robot.dIntegrateTransport_dv(s.q, dq_, sc_jacobian.Pq(), sc_jacobian.Phiv());
60 sc_jacobian.Phia() = (dt1*dt2) * sc_jacobian.Phiv();
61 sc_jacobian.Phiv().array() *= (dt1+dt2);
62 }
63 else {
64 sc_jacobian.Phiq() = sc_jacobian.Pq();
65 sc_jacobian.Phiv() = (dt1+dt2) * sc_jacobian.Pq();
66 sc_jacobian.Phia() = (dt1*dt2) * sc_jacobian.Pq();
67 }
68 kkt_residual.lx.noalias() += sc_jacobian.Phix().transpose() * s.xi_stack();
69 kkt_residual.la.noalias() += sc_jacobian.Phia().transpose() * s.xi_stack();
70}
