174inline bool ImpulseSplitKKTMatrix::isDimensionConsistent() const {
175 if (Fxx.cols() != 2*dimv_) return false;
176 if (Fxx.rows() != 2*dimv_) return false;
177 if (Qxx.cols() != 2*dimv_) return false;
178 if (Qxx.rows() != 2*dimv_) return false;
179 if (Qdvdv.cols() != dimv_) return false;
180 if (Qdvdv.rows() != dimv_) return false;
181 if (has_floating_base_) {
182 if (Fqq_prev.cols() != dimv_) return false;
183 if (Fqq_prev.rows() != dimv_) return false;
184 }
185 return true;
186}

189inline bool ImpulseSplitKKTMatrix::isApprox(
190 const ImpulseSplitKKTMatrix& other) const {
191 if (!Fxx.isApprox(other.Fxx)) return false;
192 if (!Qxx.isApprox(other.Qxx)) return false;
193 if (!Qdvdv.isApprox(other.Qdvdv)) return false;
194 if (!Qff().isApprox(other.Qff())) return false;
195 if (!Qqf().isApprox(other.Qqf())) return false;
196 if (!Fqq_prev.isApprox(other.Fqq_prev)) return false;
197 return true;
198}

212inline void ImpulseSplitKKTMatrix::setRandom() {
213 Fxx.setRandom();
214 const Eigen::MatrixXd Qxx_seed = Eigen::MatrixXd::Random(2*dimv_, 2*dimv_);
215 Qxx = Qxx_seed * Qxx_seed.transpose();
216 const Eigen::MatrixXd Qdvdvff_seed = Eigen::MatrixXd::Random(dimv_+dimi_, dimv_+dimi_);
217 const Eigen::MatrixXd Qdvdvff = Qdvdvff_seed * Qdvdvff_seed.transpose();
218 Qdvdv = Qdvdvff.topLeftCorner(dimv_, dimv_);
219 Qff() = Qdvdvff.bottomRightCorner(dimi_, dimi_);
220 Qqf().setRandom();
221 Fqq_prev.setRandom();
222}
