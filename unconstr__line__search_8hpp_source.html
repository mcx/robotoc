90 const Eigen::VectorXd& v, const Solution& s,
91 const Direction& d,
92 const double max_primal_step_size) {
93 assert(max_primal_step_size > 0);
94 assert(max_primal_step_size <= 1);
95 // If filter is empty, augment the current solution to the filter.
96 if (filter_.isEmpty()) {
97 computeCostAndViolation(ocp, robots, t, q, v, s);
98 filter_.augment(totalCosts(), totalViolations());
99 }
100 double primal_step_size = max_primal_step_size;
101 while (primal_step_size > min_step_size_) {
102 computeSolutionTrial(s, d, primal_step_size);
103 computeCostAndViolation(ocp, robots, t, q, v, s_trial_, primal_step_size);
104 const double total_costs = totalCosts();
105 const double total_violations = totalViolations();
106 if (filter_.isAccepted(total_costs, total_violations)) {
107 filter_.augment(total_costs, total_violations);
108 break;
109 }
110 primal_step_size *= step_size_reduction_rate_;
111 }
112 if (primal_step_size > min_step_size_) {
113 return primal_step_size;
114 }
115 else {
116 return min_step_size_;
117 }
118 }
