218inline void SplitSolution::integrate(const Robot& robot, const double step_size,
219 const SplitDirection& d) {
220 assert(f_stack().size() == d.df().size());
221 assert(mu_stack().size() == d.dmu().size());
222 robot.integrateConfiguration(d.dq(), step_size, q);
223 v.noalias() += step_size * d.dv();
224 a.noalias() += step_size * d.da();
225 u.noalias() += step_size * d.du;
226 lmd.noalias() += step_size * d.dlmd();
227 gmm.noalias() += step_size * d.dgmm();
228 beta.noalias() += step_size * d.dbeta();
229 if (has_floating_base_) {
230 nu_passive.noalias() += step_size * d.dnu_passive;
231 }
232 if (has_active_contacts_) {
233 f_stack().noalias() += step_size * d.df();
234 set_f_vector();
235 mu_stack().noalias() += step_size * d.dmu();
236 set_mu_vector();
237 }
238 if (has_active_impulse_) {
239 assert(xi_stack().size() == d.dxi().size());
240 xi_stack().noalias() += step_size * d.dxi();
241 }
242}

277inline bool SplitSolution::isApprox(const SplitSolution& other) const {
278 if (!q.isApprox(other.q)) {
279 return false;
280 }
281 if (!v.isApprox(other.v)) {
282 return false;
283 }
284 if (!a.isApprox(other.a)) {
285 return false;
286 }
287 if (!u.isApprox(other.u)) {
288 return false;
289 }
290 if (!lmd.isApprox(other.lmd)) {
291 return false;
292 }
293 if (!gmm.isApprox(other.gmm)) {
294 return false;
295 }
296 if (!beta.isApprox(other.beta)) {
297 return false;
298 }
299 if (has_floating_base_) {
300 if (!nu_passive.isApprox(other.nu_passive)) {
301 return false;
302 }
303 }
304 if (has_active_contacts_) {
305 if (!f_stack().isApprox(other.f_stack())) {
306 return false;
307 }
308 if (!mu_stack().isApprox(other.mu_stack())) {
309 return false;
310 }
311 for (int i=0; i<is_contact_active_.size(); ++i) {
312 if (is_contact_active_[i]) {
313 if (!other.isContactActive(i)) {
314 return false;
315 }
316 if (!f[i].isApprox(other.f[i])) {
317 return false;
318 }
319 if (!mu[i].isApprox(other.mu[i])) {
320 return false;
321 }
322 }
323 else {
324 if (other.isContactActive(i)) {
325 return false;
326 }
327 }
328 }
329 }
330 if (has_active_impulse_) {
331 if (!xi_stack().isApprox(other.xi_stack())) {
332 return false;
333 }
334 }
335 return true;
336}
