37inline void TerminalStateEquation::linearizeStateEquation(
38 const Robot& robot, const Eigen::MatrixBase<ConfigVectorType>& q_prev,
39 const SplitSolution& s, SplitKKTMatrix& kkt_matrix,
40 SplitKKTResidual& kkt_residual) {
41 assert(q_prev.size() == robot.dimq());
42 if (robot.hasFloatingBase()) {
43 robot.dSubtractConfiguration_dq0(q_prev, s.q, kkt_matrix.Fqq_prev);
44 kkt_residual.lq().template head<6>().noalias()
45 += kkt_matrix.Fqq_prev.template topLeftCorner<6, 6>().transpose()
46 * s.lmd.template head<6>();
47 kkt_residual.lq().tail(robot.dimv()-6).noalias()
48 -= s.lmd.tail(robot.dimv()-6);
49 }
50 else {
51 kkt_residual.lq().noalias() -= s.lmd;
52 }
53 kkt_residual.lv().noalias() -= s.gmm;
54}
