158inline void ImpulseSplitSolution::integrate(const Robot& robot,
159 const double step_size,
160 const ImpulseSplitDirection& d) {
161 assert(f_stack().size() == d.df().size());
162 assert(mu_stack().size() == d.dmu().size());
163 robot.integrateConfiguration(d.dq(), step_size, q);
164 v.noalias() += step_size * d.dv();
165 dv.noalias() += step_size * d.ddv();
166 f_stack().noalias() += step_size * d.df();
167 set_f_vector();
168 lmd.noalias() += step_size * d.dlmd();
169 gmm.noalias() += step_size * d.dgmm();
170 beta.noalias() += step_size * d.dbeta();
171 mu_stack().noalias() += step_size * d.dmu();
172 set_mu_vector();
173}

200inline bool ImpulseSplitSolution::isApprox(
201 const ImpulseSplitSolution& other) const {
202 if (!q.isApprox(other.q)) {
203 return false;
204 }
205 if (!v.isApprox(other.v)) {
206 return false;
207 }
208 if (!dv.isApprox(other.dv)) {
209 return false;
210 }
211 if (!f_stack().isApprox(other.f_stack())) {
212 return false;
213 }
214 if (!lmd.isApprox(other.lmd)) {
215 return false;
216 }
217 if (!gmm.isApprox(other.gmm)) {
218 return false;
219 }
220 if (!beta.isApprox(other.beta)) {
221 return false;
222 }
223 if (!mu_stack().isApprox(other.mu_stack())) {
224 return false;
225 }
226 for (int i=0; i<is_impulse_active_.size(); ++i) {
227 if (is_impulse_active_[i]) {
228 if (!other.isImpulseActive(i)) {
229 return false;
230 }
231 if (!f[i].isApprox(other.f[i])) {
232 return false;
233 }
234 if (!mu[i].isApprox(other.mu[i])) {
235 return false;
236 }
237 }
238 else {
239 if (other.isImpulseActive(i)) {
240 return false;
241 }
242 }
243 }
244 return true;
245}
