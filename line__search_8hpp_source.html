121 static void computeSolutionTrial(const Robot& robot, const SplitSolution& s,
122 const SplitDirection& d,
123 const double step_size,
124 SplitSolution& s_trial) {
125 s_trial.setContactStatus(s);
126 robot.integrateConfiguration(s.q, d.dq(), step_size, s_trial.q);
127 s_trial.v = s.v + step_size * d.dv();
128 s_trial.a = s.a + step_size * d.da();
129 s_trial.u = s.u + step_size * d.du;
130 if (s.hasActiveContacts()) {
131 s_trial.f_stack() = s.f_stack() + step_size * d.df();
132 s_trial.set_f_vector();
133 }
134 }
