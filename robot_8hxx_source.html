22template <typename TangentVectorType, typename ConfigVectorType>
23inline void Robot::integrateConfiguration(
24 const Eigen::MatrixBase<TangentVectorType>& v,
25 const double integration_length,
26 const Eigen::MatrixBase<ConfigVectorType>& q) const {
27 assert(v.size() == dimv_);
28 assert(q.size() == dimq_);
29 if (has_floating_base_) {
30 const Eigen::VectorXd q_tmp = q;
31 pinocchio::integrate(model_, q_tmp, integration_length*v,
32 const_cast<Eigen::MatrixBase<ConfigVectorType>&>(q));
33 }
34 else {
35 (const_cast<Eigen::MatrixBase<ConfigVectorType>&>(q)).noalias()
36 += integration_length * v;
37 }
38}

242 .template segment<3>(3*num_active_contacts));
243 ++num_active_contacts;
244 }
245 }

504template <typename MatrixType1, typename MatrixType2, typename MatrixType3>
505inline void Robot::computeMJtJinv(
506 const Eigen::MatrixBase<MatrixType1>& M,
507 const Eigen::MatrixBase<MatrixType2>& J,
508 const Eigen::MatrixBase<MatrixType3>& MJtJinv) {
509 assert(M.rows() == dimv_);
510 assert(M.cols() == dimv_);
511 assert(J.rows() <= max_dimf_);
512 assert(J.cols() == dimv_);
513 assert(MJtJinv.rows() == M.rows()+J.rows());
514 assert(MJtJinv.cols() == M.rows()+J.rows());
515 const int dimf = J.rows();
516 data_.M = M;
517 pinocchio::cholesky::decompose(model_, data_);
518 data_.sDUiJt.leftCols(dimf) = J.transpose();
519 pinocchio::cholesky::Uiv(model_, data_, data_.sDUiJt.leftCols(dimf));
520 for (Eigen::DenseIndex k=0; k<dimv_; ++k) {
521 data_.sDUiJt.leftCols(dimf).row(k) /= std::sqrt(data_.D[k]);
522 }
523 data_.JMinvJt.topLeftCorner(dimf, dimf).noalias()
524 = data_.sDUiJt.leftCols(dimf).transpose() * data_.sDUiJt.leftCols(dimf);
525 data_.llt_JMinvJt.compute(data_.JMinvJt.topLeftCorner(dimf, dimf));
526 assert(data_.llt_JMinvJt.info() == Eigen::Success);
527 Eigen::Block<MatrixType3> topLeft
528 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).topLeftCorner(dimv_, dimv_);
529 Eigen::Block<MatrixType3> topRight
530 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).topRightCorner(dimv_, dimf);
531 Eigen::Block<MatrixType3> bottomLeft
532 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).bottomLeftCorner(dimf, dimv_);
533 Eigen::Block<MatrixType3> bottomRight
534 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).bottomRightCorner(dimf, dimf);
535 bottomRight = - pinocchio::Data::MatrixXs::Identity(dimf, dimf);
536 topLeft.setIdentity();
537 data_.llt_JMinvJt.solveInPlace(bottomRight);
538 pinocchio::cholesky::solve(model_, data_, topLeft);
539 bottomLeft.noalias() = J * topLeft;
540 topRight.noalias() = bottomLeft.transpose() * (-bottomRight);
541 topLeft.noalias() -= topRight*bottomLeft;
542 bottomLeft = topRight.transpose();
543 assert(!MJtJinv.hasNaN());
544}

547inline Eigen::VectorXd Robot::generateFeasibleConfiguration() const {
548 Eigen::VectorXd q_min(dimq_), q_max(dimq_);
549 if (has_floating_base_) {
550 q_min.head(7) = - Eigen::VectorXd::Ones(7);
551 q_max.head(7) = Eigen::VectorXd::Ones(7);
552 }
553 q_min.tail(dimu_) = lower_joint_position_limit_;
554 q_max.tail(dimu_) = upper_joint_position_limit_;
555 return pinocchio::randomConfiguration(model_, q_min, q_max);
556}

559template <typename ConfigVectorType>
560inline void Robot::normalizeConfiguration(
561 const Eigen::MatrixBase<ConfigVectorType>& q) const {
562 assert(q.size() == dimq_);
563 if (has_floating_base_) {
564 if (q.template segment<4>(3).squaredNorm()
565 <= std::numeric_limits<double>::epsilon()) {
566 (const_cast<Eigen::MatrixBase<ConfigVectorType>&> (q)).coeffRef(3) = 1;
567 }
568 pinocchio::normalize(model_,
569 const_cast<Eigen::MatrixBase<ConfigVectorType>&>(q));
570 }
