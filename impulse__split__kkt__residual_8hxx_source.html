150inline bool ImpulseSplitKKTResidual::isApprox(
151 const ImpulseSplitKKTResidual& other) const {
152 assert(isDimensionConsistent());
153 assert(other.isDimensionConsistent());
154 if (!Fx.isApprox(other.Fx)) return false;
155 if (!lx.isApprox(other.lx)) return false;
156 if (!ldv.isApprox(other.ldv)) return false;
157 if (!lf().isApprox(other.lf())) return false;
158 Eigen::VectorXd vec(3), other_vec(3);
159 vec << kkt_error, cost, constraint_violation;
160 other_vec << other.kkt_error, other.cost, other.constraint_violation;
161 if (!vec.isApprox(other_vec)) return false;
162 return true;
163}
