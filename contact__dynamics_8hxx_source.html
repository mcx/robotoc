34inline void ContactDynamics::evalContactDynamics(
35 Robot& robot, const ContactStatus& contact_status, const SplitSolution& s) {
36 data_.setContactStatus(contact_status);
37 has_active_contacts_ = contact_status.hasActiveContacts();
38 robot.setContactForces(contact_status, s.f);
39 robot.RNEA(s.q, s.v, s.a, data_.ID_full());
40 data_.ID().noalias() -= s.u;
41 robot.computeBaumgarteResidual(contact_status, contact_status.contactPoints(),
42 data_.C());
43}

46inline void ContactDynamics::linearizeContactDynamics(
47 Robot& robot, const ContactStatus& contact_status, const double dt,
48 const SplitSolution& s, SplitKKTResidual& kkt_residual) {
49 assert(dt > 0);
50 evalContactDynamics(robot, contact_status, s);
51 robot.RNEADerivatives(s.q, s.v, s.a, data_.dIDdq(), data_.dIDdv(), data_.dIDda);
52 robot.computeBaumgarteDerivatives(contact_status, data_.dCdq(), data_.dCdv(),
53 data_.dCda());
54 // augment inverse dynamics constraint
55 kkt_residual.lq().noalias() += dt * data_.dIDdq().transpose() * s.beta;
56 kkt_residual.lv().noalias() += dt * data_.dIDdv().transpose() * s.beta;
57 kkt_residual.la.noalias() += dt * data_.dIDda.transpose() * s.beta;
58 if (has_active_contacts_) {
59 kkt_residual.lf().noalias() -= dt * data_.dCda() * s.beta;
60 }
61 if (has_floating_base_) {
62 // augment floating base constraint
63 data_.lu_passive = dt * s.nu_passive;
64 data_.lu_passive.noalias() -= dt * s.beta.template head<kDimFloatingBase>();
65 kkt_residual.lu.noalias() -= dt * s.beta.tail(robot.dimu());
66 }
67 else {
68 kkt_residual.lu.noalias() -= dt * s.beta;
69 }
70 // augment acceleration-level contact constraint
71 if (has_active_contacts_) {
72 kkt_residual.lq().noalias()
73 += dt * data_.dCdq().transpose() * s.mu_stack();
74 kkt_residual.lv().noalias()
75 += dt * data_.dCdv().transpose() * s.mu_stack();
76 kkt_residual.la.noalias() += dt * data_.dCda().transpose() * s.mu_stack();
77 }
78}

81inline void ContactDynamics::condenseContactDynamics(
82 Robot& robot, const ContactStatus& contact_status, const double dt,
83 const SplitSolution& s, SplitKKTMatrix& kkt_matrix,
84 SplitKKTResidual& kkt_residual) {
85 assert(dt > 0);
86 const int dimv = robot.dimv();
87 const int dimu = robot.dimu();
88 const int dim_passive = robot.dim_passive();
89 const int dimf = contact_status.dimf();
90 robot.computeMJtJinv(data_.dIDda, data_.dCda(), data_.MJtJinv());
91 data_.MJtJinv_dIDCdqv().noalias() = data_.MJtJinv() * data_.dIDCdqv();
92 data_.MJtJinv_IDC().noalias() = data_.MJtJinv() * data_.IDC();
93
94 data_.Qafqv().topRows(dimv).noalias()
95 = (- kkt_matrix.Qaa.diagonal()).asDiagonal()
96 * data_.MJtJinv_dIDCdqv().topRows(dimv);
97 data_.Qafqv().bottomRows(dimf).noalias()
98 = - kkt_matrix.Qff() * data_.MJtJinv_dIDCdqv().bottomRows(dimf);
99 data_.Qafqv().bottomLeftCorner(dimf, dimv).noalias()
100 -= kkt_matrix.Qqf().transpose();
101 data_.Qafu_full().topRows(dimv).noalias()
102 = kkt_matrix.Qaa.diagonal().asDiagonal()
103 * data_.MJtJinv().topLeftCorner(dimv, dimv);
104 data_.Qafu_full().bottomRows(dimf).noalias()
105 = kkt_matrix.Qff() * data_.MJtJinv().bottomLeftCorner(dimf, dimv);
106 data_.la() = kkt_residual.la;
107 data_.lf() = - kkt_residual.lf();
108 data_.la().noalias()
109 -= kkt_matrix.Qaa.diagonal().asDiagonal()
110 * data_.MJtJinv_IDC().head(dimv);
111 data_.lf().noalias()
112 -= kkt_matrix.Qff() * data_.MJtJinv_IDC().tail(dimf);
113
114 kkt_matrix.Qxx.noalias()
115 -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafqv();
116 kkt_matrix.Qxx.topRows(dimv).noalias()
117 += kkt_matrix.Qqf() * data_.MJtJinv_dIDCdqv().bottomRows(dimf);
118 if (has_floating_base_) {
119 data_.Qxu_passive.noalias()
120 = - data_.MJtJinv_dIDCdqv().transpose() * data_.Qafu_full().leftCols(dim_passive);
121 data_.Qxu_passive.topRows(dimv).noalias()
122 -= kkt_matrix.Qqf() * data_.MJtJinv().bottomLeftCorner(dimf, dimv).leftCols(dim_passive);
123 kkt_matrix.Qxu.noalias()
124 -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafu_full().rightCols(dimu);
125 kkt_matrix.Qxu.topRows(dimv).noalias()
126 -= kkt_matrix.Qqf() * data_.MJtJinv().bottomLeftCorner(dimf, dimv).rightCols(dimu);
127 }
128 else {
129 kkt_matrix.Qxu.noalias()
130 -= data_.MJtJinv_dIDCdqv().transpose() * data_.Qafu_full();
131 kkt_matrix.Qxu.topRows(dimv).noalias()
132 -= kkt_matrix.Qqf() * data_.MJtJinv().bottomLeftCorner(dimf, dimv);
133 }
134 kkt_residual.lx.noalias()
135 -= data_.MJtJinv_dIDCdqv().transpose() * data_.laf();
136 kkt_residual.lq().noalias()
137 += kkt_matrix.Qqf() * data_.MJtJinv_IDC().tail(dimf);
138
139 if (has_floating_base_) {
140 data_.Quu_passive_topRight.noalias()
141 = data_.MJtJinv().topRows(dim_passive) * data_.Qafu_full().rightCols(dimu);
142 kkt_matrix.Quu.noalias()
143 += data_.MJtJinv().middleRows(dim_passive, dimu) * data_.Qafu_full().rightCols(dimu);
144 }
145 else {
146 kkt_matrix.Quu.noalias()
147 += data_.MJtJinv().topRows(dimv) * data_.Qafu_full();
148 }
149 if (has_floating_base_) {
150 data_.lu_passive.noalias()
151 += data_.MJtJinv().template topRows<kDimFloatingBase>() * data_.laf();
152 }
153 kkt_residual.lu.noalias()
154 += data_.MJtJinv().middleRows(dim_passive, dimu) * data_.laf();
155
156 kkt_matrix.Fvq() = - dt * data_.MJtJinv_dIDCdqv().topLeftCorner(dimv, dimv);
157 kkt_matrix.Fvv().noalias()
158 = - dt * data_.MJtJinv_dIDCdqv().topRightCorner(dimv, dimv)
159 + Eigen::MatrixXd::Identity(dimv, dimv);
160 kkt_matrix.Fvu = dt * data_.MJtJinv().block(0, dim_passive, dimv, dimu);
161 kkt_residual.Fv().noalias() -= dt * data_.MJtJinv_IDC().head(dimv);
162}

165inline void ContactDynamics::expandPrimal(SplitDirection& d) const {
166 d.daf().noalias() = - data_.MJtJinv_dIDCdqv() * d.dx;
167 d.daf().noalias()
168 += data_.MJtJinv().middleCols(dim_passive_, dimu_) * d.du;
169 d.daf().noalias() -= data_.MJtJinv_IDC();
170 d.df().array() *= -1;
171}

185template <typename SplitDirectionType>
186inline void ContactDynamics::expandDual(const double dt,
187 const SplitDirectionType& d_next,
188 SplitDirection& d) {
189 assert(dt > 0);
190 if (has_floating_base_) {
191 d.dnu_passive = data_.lu_passive;
192 d.dnu_passive.noalias() += data_.Quu_passive_topRight * d.du;
193 d.dnu_passive.noalias() += data_.Qxu_passive.transpose() * d.dx;
194 d.dnu_passive.noalias()
195 += dt * data_.MJtJinv().leftCols(dimv_).template topRows<kDimFloatingBase>()
196 * d_next.dgmm();
197 d.dnu_passive.array() *= - (1/dt);
198 }
199 data_.laf().noalias() += data_.Qafqv() * d.dx;
200 data_.laf().noalias() += data_.Qafu() * d.du;
201 data_.la().noalias() += dt * d_next.dgmm();
202 d.dbetamu().noalias() = - data_.MJtJinv() * data_.laf() * (1/dt);
203}

207inline void ContactDynamics::expandDual(const double dt, const double dts,
208 const SplitDirectionType& d_next,
209 SplitDirection& d) {

212 d.dnu_passive = (1+(dts/dt)) * data_.lu_passive;

214 d.dnu_passive.noalias() += data_.Qxu_passive.transpose() * d.dx;
215 d.dnu_passive.noalias()

222 data_.laf().noalias() += data_.Qafu() * d.du;
