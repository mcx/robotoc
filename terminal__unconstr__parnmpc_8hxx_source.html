12inline TerminalUnconstrParNMPC::TerminalUnconstrParNMPC(
13 const Robot& robot, const std::shared_ptr<CostFunction>& cost,
14 const std::shared_ptr<Constraints>& constraints)
15 : cost_(cost),
16 cost_data_(cost->createCostFunctionData(robot)),
17 constraints_(constraints),
18 constraints_data_(constraints->createConstraintsData(robot, 0)),
19 unconstr_dynamics_(robot),
20 use_kinematics_(false),
21 stage_cost_(0) {
22 if (cost_->useKinematics() || constraints_->useKinematics()) {
23 use_kinematics_ = true;
24 }
25 try {
26 if (robot.hasFloatingBase()) {
27 throw std::logic_error(
28 "robot has floating base: robot should have no constraints!");
29 }
30 if (robot.maxPointContacts() > 0) {
31 throw std::logic_error(
32 "robot can have contacts: robot should have no constraints!");
33 }
34 }
35 catch(const std::exception& e) {
36 std::cerr << e.what() << '\n';
37 std::exit(EXIT_FAILURE);
38 }
39}

72inline void TerminalUnconstrParNMPC::evalOCP(Robot& robot, const double t,
73 const double dt,
74 const Eigen::VectorXd& q_prev,
75 const Eigen::VectorXd& v_prev,
76 const SplitSolution& s,
77 SplitKKTResidual& kkt_residual) {
78 assert(dt > 0);
79 assert(q_prev.size() == robot.dimq());
80 assert(v_prev.size() == robot.dimv());
81 if (use_kinematics_) {
82 robot.updateKinematics(s.q);
83 }
84 kkt_residual.setZero();
85 stage_cost_ = cost_->evalStageCost(robot, cost_data_, t, dt, s);
86 stage_cost_ += cost_->evalTerminalCost(robot, cost_data_, t, s);
87 constraints_->evalConstraint(robot, constraints_data_, s);
88 stage_cost_ += dt * constraints_data_.logBarrier();
89 unconstr::stateequation::computeBackwardEulerResidual(dt, q_prev, v_prev, s,
90 kkt_residual);
91 unconstr_dynamics_.evalUnconstrDynamics(robot, s);
92}

95inline void TerminalUnconstrParNMPC::computeKKTResidual(
96 Robot& robot, const double t, const double dt,
97 const Eigen::VectorXd& q_prev, const Eigen::VectorXd& v_prev,
98 const SplitSolution& s, SplitKKTMatrix& kkt_matrix,

100 assert(dt > 0);
101 assert(q_prev.size() == robot.dimq());
102 assert(v_prev.size() == robot.dimv());

104 robot.updateKinematics(s.q);

121inline void TerminalUnconstrParNMPC::computeKKTSystem(
122 Robot& robot, const double t, const double dt,
123 const Eigen::VectorXd& q_prev, const Eigen::VectorXd& v_prev,
124 const SplitSolution& s, SplitKKTMatrix& kkt_matrix,
125 SplitKKTResidual& kkt_residual) {
126 assert(dt > 0);
127 assert(q_prev.size() == robot.dimq());
128 assert(v_prev.size() == robot.dimv());
129 if (use_kinematics_) {
130 robot.updateKinematics(s.q);
131 }
132 kkt_matrix.setZero();
133 kkt_residual.setZero();
134 stage_cost_ = cost_->quadratizeStageCost(robot, cost_data_, t, dt, s,
135 kkt_residual, kkt_matrix);
136 stage_cost_ += cost_->quadratizeTerminalCost(robot, cost_data_, t, s,
137 kkt_residual, kkt_matrix);
138 constraints_->linearizeConstraints(robot, constraints_data_, dt, s,
139 kkt_residual);
140 stage_cost_ += dt * constraints_data_.logBarrier();
141 unconstr::stateequation::linearizeBackwardEulerTerminal(dt, q_prev, v_prev, s,
142 kkt_matrix, kkt_residual);
143 unconstr_dynamics_.linearizeUnconstrDynamics(robot, dt, s, kkt_residual);
144 kkt_residual.kkt_error = KKTError(kkt_residual, dt);
145 constraints_->condenseSlackAndDual(constraints_data_, dt, s,
146 kkt_matrix, kkt_residual);
147 unconstr_dynamics_.condenseUnconstrDynamics(kkt_matrix, kkt_residual);
148}

189inline double TerminalUnconstrParNMPC::KKTError(
190 const SplitKKTResidual& kkt_residual, const double dt) const {
191 assert(dt > 0);
192 double err = 0;
193 err += kkt_residual.KKTError();
194 err += (dt*dt) * unconstr_dynamics_.KKTError();
195 err += (dt*dt) * constraints_data_.KKTError();
196 return err;
197}
