61inline void ImpulseSplitOCP::evalOCP(Robot& robot,
62 const ImpulseStatus& impulse_status,
63 const double t,
64 const ImpulseSplitSolution& s,
65 const Eigen::VectorXd& q_next,
66 const Eigen::VectorXd& v_next,
67 ImpulseSplitKKTResidual& kkt_residual) {
68 assert(q_next.size() == robot.dimq());
69 assert(v_next.size() == robot.dimv());
70 kkt_residual.setImpulseStatus(impulse_status);
71 kkt_residual.setZero();
72 robot.updateKinematics(s.q, s.v+s.dv);
73 stage_cost_ = cost_->evalImpulseCost(robot, cost_data_, t, s);
74 constraints_->evalConstraint(robot, constraints_data_, s);
75 stage_cost_ += constraints_data_.logBarrier();
76 state_equation_.evalStateEquation(robot, s, q_next, v_next, kkt_residual);
77 impulse_dynamics_.evalImpulseDynamics(robot, impulse_status, s);
78}

104inline void ImpulseSplitOCP::computeKKTSystem(
105 Robot& robot, const ImpulseStatus& impulse_status, const double t,
106 const Eigen::VectorXd& q_prev, const ImpulseSplitSolution& s,
107 const SplitSolution& s_next, ImpulseSplitKKTMatrix& kkt_matrix,
108 ImpulseSplitKKTResidual& kkt_residual) {
109 assert(q_prev.size() == robot.dimq());
110 robot.updateKinematics(s.q, s.v+s.dv);
111 kkt_matrix.setImpulseStatus(impulse_status);
112 kkt_residual.setImpulseStatus(impulse_status);
113 kkt_matrix.setZero();
114 kkt_residual.setZero();
115 stage_cost_ = cost_->quadratizeImpulseCost(robot, cost_data_, t, s,
116 kkt_residual, kkt_matrix);
117 constraints_->linearizeConstraints(robot, constraints_data_, s, kkt_residual);
118 stage_cost_ += constraints_data_.logBarrier();
119 state_equation_.linearizeStateEquation(robot, q_prev, s, s_next,
120 kkt_matrix, kkt_residual);
121 impulse_dynamics_.linearizeImpulseDynamics(robot, impulse_status, s,
122 kkt_residual);
123 kkt_residual.kkt_error = KKTError(kkt_residual);
124 constraints_->condenseSlackAndDual(constraints_data_, s,
125 kkt_matrix, kkt_residual);
126 impulse_dynamics_.condenseImpulseDynamics(robot, impulse_status,
127 kkt_matrix, kkt_residual);
128 state_equation_.correctLinearizedStateEquation(robot, s, s_next,
129 kkt_matrix, kkt_residual);
