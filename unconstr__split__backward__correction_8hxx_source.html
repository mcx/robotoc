43template <typename MatrixType>
44inline void UnconstrSplitBackwardCorrection::coarseUpdate(
45 const Eigen::MatrixBase<MatrixType>& aux_mat_next, const double dt,
46 const SplitKKTMatrix& kkt_matrix, const SplitKKTResidual& kkt_residual,
47 const SplitSolution& s, SplitSolution& s_new) {
48 assert(aux_mat_next.rows() == dimx_);
49 assert(aux_mat_next.cols() == dimx_);
50 H_.topLeftCorner(dimv_, dimv_) = kkt_matrix.Qaa;
51 H_.bottomLeftCorner(dimx_, dimv_) = kkt_matrix.Qxu; // This is actually Qxa
52 H_.bottomRightCorner(dimx_, dimx_) = kkt_matrix.Qxx.transpose();
53 H_.bottomRightCorner(dimx_, dimx_).noalias() += aux_mat_next;
54 kkt_mat_inverter_.invert(dt, H_, kkt_mat_inv_);
55 kkt_res_.head(dimx_) = kkt_residual.Fx;
56 kkt_res_.segment(dimx_, dimv_) = kkt_residual.la;
57 kkt_res_.tail(dimx_) = kkt_residual.lx;
58 d_.noalias() = kkt_mat_inv_ * kkt_res_;
59
60 s_new.lmd = s.lmd - d_.head(dimv_);
61 s_new.gmm = s.gmm - d_.segment(dimv_, dimv_);
62 s_new.a = s.a - d_.segment(2*dimv_, dimv_);
63 s_new.q = s.q - d_.segment(3*dimv_, dimv_);
64 s_new.v = s.v - d_.tail(dimv_);
65}
