63inline void UnconstrDynamics::linearizeUnconstrDynamics(
64 Robot& robot, const double dt, const SplitSolution& s,
65 SplitKKTResidual& kkt_residual) {
66 assert(dt > 0);
67 evalUnconstrDynamics(robot, s);
68 // augment inverse dynamics constraint
69 robot.RNEADerivatives(s.q, s.v, s.a, dID_dq_, dID_dv_, dID_da_);
70 kkt_residual.lq().noalias() += dt * dID_dq_.transpose() * s.beta;
71 kkt_residual.lv().noalias() += dt * dID_dv_.transpose() * s.beta;
72 kkt_residual.la.noalias() += dt * dID_da_.transpose() * s.beta;
73 kkt_residual.lu.noalias() -= dt * s.beta;
74}

77inline void UnconstrDynamics::condenseUnconstrDynamics(
78 SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual) {
79 // condense KKT residual
80 lu_condensed_ = kkt_residual.lu;
81 lu_condensed_.noalias() += kkt_matrix.Quu.diagonal().asDiagonal() * ID_;
82 kkt_residual.lq().noalias() += dID_dq_.transpose() * lu_condensed_;
83 kkt_residual.lv().noalias() += dID_dv_.transpose() * lu_condensed_;
84 kkt_residual.la.noalias() += dID_da_.transpose() * lu_condensed_;
85 // condense KKT Hessian
86 Quu_dID_dq_.noalias() = kkt_matrix.Quu.diagonal().asDiagonal() * dID_dq_;
87 Quu_dID_dv_.noalias() = kkt_matrix.Quu.diagonal().asDiagonal() * dID_dv_;
88 Quu_dID_da_.noalias() = kkt_matrix.Quu.diagonal().asDiagonal() * dID_da_;
89 kkt_matrix.Qqq().noalias() += dID_dq_.transpose() * Quu_dID_dq_;
90 kkt_matrix.Qqv().noalias() += dID_dq_.transpose() * Quu_dID_dv_;
91 kkt_matrix.Qvq() = kkt_matrix.Qqv().transpose();
92 kkt_matrix.Qvv().noalias() += dID_dv_.transpose() * Quu_dID_dv_;
93 kkt_matrix.Qaa.noalias() += dID_da_.transpose() * Quu_dID_da_;
94 // this is actually Qqa and Qqv
95 kkt_matrix.Qqu().noalias() = Quu_dID_dq_.transpose() * dID_da_;
96 kkt_matrix.Qvu().noalias() = Quu_dID_dv_.transpose() * dID_da_;
97}

100inline void UnconstrDynamics::expandPrimal(SplitDirection& d) const {
101 d.du = ID_;

110 const SplitKKTMatrix& kkt_matrix,
111 const SplitKKTResidual& kkt_residual,
112 SplitDirection& d) {
113 assert(dt > 0);
