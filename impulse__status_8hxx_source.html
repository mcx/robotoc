70inline void ImpulseStatus::setActivity(
71 const ContactStatus& pre_contact_status,
72 const ContactStatus& post_contact_status) {
73 assert(pre_contact_status.maxPointContacts() == maxPointContacts());
74 assert(post_contact_status.maxPointContacts() == maxPointContacts());
75 for (int i=0; i<contact_status_.maxPointContacts(); ++i) {

77 deactivateImpulse(i);

80 if (post_contact_status.isContactActive(i)) {
