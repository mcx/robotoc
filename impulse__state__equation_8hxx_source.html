57template <typename ConfigVectorType>
58inline void ImpulseStateEquation::linearizeStateEquation(
59 const Robot& robot, const Eigen::MatrixBase<ConfigVectorType>& q_prev,
60 const ImpulseSplitSolution& s, const SplitSolution& s_next,
61 ImpulseSplitKKTMatrix& kkt_matrix, ImpulseSplitKKTResidual& kkt_residual) {
62 assert(q_prev.size() == robot.dimq());
63 evalStateEquation(robot, s, s_next.q, s_next.v, kkt_residual);
64 if (robot.hasFloatingBase()) {
65 robot.dSubtractConfiguration_dqf(s.q, s_next.q, kkt_matrix.Fqq());
66 robot.dSubtractConfiguration_dq0(q_prev, s.q, kkt_matrix.Fqq_prev);
67 kkt_residual.lq().template head<6>().noalias()
68 += kkt_matrix.Fqq().template topLeftCorner<6, 6>().transpose()
69 * s_next.lmd.template head<6>();
70 kkt_residual.lq().template head<6>().noalias()
71 += kkt_matrix.Fqq_prev.template topLeftCorner<6, 6>().transpose()
72 * s.lmd.template head<6>();
73 kkt_residual.lq().tail(robot.dimv()-6).noalias()
74 += s_next.lmd.tail(robot.dimv()-6) - s.lmd.tail(robot.dimv()-6);
75 }
76 else {
77 kkt_matrix.Fqq().diagonal().fill(1.);
78 kkt_residual.lq().noalias() += s_next.lmd - s.lmd;
79 }
80 kkt_residual.lv().noalias() += s_next.gmm - s.gmm;
81 kkt_residual.ldv.noalias() += s_next.gmm;
82}

85inline void ImpulseStateEquation::correctLinearizedStateEquation(
86 const Robot& robot, const ImpulseSplitSolution& s,
87 const SplitSolution& s_next, ImpulseSplitKKTMatrix& kkt_matrix,
88 ImpulseSplitKKTResidual& kkt_residual) {
89 if (has_floating_base_) {
90 se3_jac_inverse_.compute(kkt_matrix.Fqq_prev, Fqq_prev_inv_);
91 robot.dSubtractConfiguration_dq0(s.q, s_next.q, kkt_matrix.Fqq_prev);
92 se3_jac_inverse_.compute(kkt_matrix.Fqq_prev, Fqq_inv_);
93 Fqq_tmp_ = kkt_matrix.Fqq().template topLeftCorner<6, 6>();
94 Fq_tmp_ = kkt_residual.Fq().template head<6>();
95 kkt_matrix.Fqq().template topLeftCorner<6, 6>().noalias() = - Fqq_inv_ * Fqq_tmp_;
96 kkt_residual.Fq().template head<6>().noalias() = - Fqq_inv_ * Fq_tmp_;
97 }
