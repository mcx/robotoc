146inline bool SwitchingConstraintJacobian::isApprox(

148 assert(dimi() == other.dimi());
149 if (!Pq().isApprox(other.Pq())) return false;
150 if (!Phix().isApprox(other.Phix())) return false;
151 if (!Phia().isApprox(other.Phia())) return false;
