46inline void UnconstrRiccatiFactorizer::forwardRiccatiRecursion(
47 const SplitKKTResidual& kkt_residual, const double dt,
48 const LQRPolicy& lqr_policy, SplitDirection& d,
49 SplitDirection& d_next) const {
50 assert(dt > 0);
51 d.da().noalias() = lqr_policy.K * d.dx + lqr_policy.k;
52 d_next.dx = kkt_residual.Fx + d.dx;
53 d_next.dq().noalias() += dt * d.dv();
54 d_next.dv().noalias() += dt * d.da();
