225inline bool SplitDirection::isDimensionConsistent() const {
226 if (dx.size() != 2*dimv_) return false;
227 if (du.size() != dimu_) return false;
228 if (dlmdgmm.size() != 2*dimv_) return false;
229 if (dnu_passive.size() != dim_passive_) return false;
230 return true;
231}

234inline bool SplitDirection::isApprox(const SplitDirection& other) const {
235 assert(isDimensionConsistent());
236 assert(other.isDimensionConsistent());
237 assert(dimf()==other.dimf());
238 assert(dimi()==other.dimi());
239 if (!dx.isApprox(other.dx)) {
240 return false;
241 }
242 if (!du.isApprox(other.du)) {
243 return false;
244 }
245 if (!daf().isApprox(other.daf())) {
246 return false;
247 }
248 if (!dlmdgmm.isApprox(other.dlmdgmm)) {
249 return false;
250 }
251 if (!dbetamu().isApprox(other.dbetamu())) {
252 return false;
253 }
254 if (!dxi().isApprox(other.dxi())) {
255 return false;
256 }
257 if (!dnu_passive.isApprox(other.dnu_passive)) {
258 return false;
259 }
260 Eigen::VectorXd vec(1), other_vec(1);
261 vec << dts;
262 other_vec << other.dts;
263 if (!vec.isApprox(other_vec)) {
264 return false;
265 }
266 return true;
267}
