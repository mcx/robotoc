190inline bool ImpulseSplitDirection::isApprox(
191 const ImpulseSplitDirection& other) const {
192 assert(isDimensionConsistent());
193 assert(other.isDimensionConsistent());
194 assert(dimi()==other.dimi());
195 if (!dx.isApprox(other.dx)) return false;
196 if (!ddvf().isApprox(other.ddvf())) return false;
197 if (!dlmdgmm.isApprox(other.dlmdgmm)) return false;
198 if (!dbetamu().isApprox(other.dbetamu())) return false;
199 return true;
200}
