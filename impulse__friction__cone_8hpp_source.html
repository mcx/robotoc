127 template <typename VectorType1, typename VectorType2>
128 static void frictionConeResidual(const double mu,
129 const Eigen::MatrixBase<VectorType1>& f,
130 const Eigen::MatrixBase<VectorType2>& res) {
131 assert(mu > 0);
132 assert(f.size() == 3);
133 assert(res.size() == 5);
134 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(0) = - f.coeff(2);
135 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(1)
136 = f.coeff(0) - mu * f.coeff(2) / std::sqrt(2);
137 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(2)
138 = - f.coeff(0) - mu * f.coeff(2) / std::sqrt(2);
139 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(3)
140 = f.coeff(1) - mu * f.coeff(2) / std::sqrt(2);
141 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(4)
142 = - f.coeff(1) - mu * f.coeff(2) / std::sqrt(2);
143 }
