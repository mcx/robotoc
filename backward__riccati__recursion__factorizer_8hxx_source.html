32inline void BackwardRiccatiRecursionFactorizer::factorizeKKTMatrix(
33 const SplitRiccatiFactorization& riccati_next,
34 SplitKKTMatrix& kkt_matrix, SplitKKTResidual& kkt_residual) {
35 AtP_.noalias() = kkt_matrix.Fxx.transpose() * riccati_next.P;
36 BtP_.noalias() = kkt_matrix.Fvu.transpose() * riccati_next.P.bottomRows(dimv_);
37 // Factorize F
38 kkt_matrix.Qxx.noalias() += AtP_ * kkt_matrix.Fxx;
39 // Factorize H
40 kkt_matrix.Qxu.noalias() += AtP_.rightCols(dimv_) * kkt_matrix.Fvu;
41 // Factorize G
42 kkt_matrix.Quu.noalias() += BtP_.rightCols(dimv_) * kkt_matrix.Fvu;
43 // Factorize vector term
44 kkt_residual.lu.noalias() += BtP_ * kkt_residual.Fx;
45 kkt_residual.lu.noalias() -= kkt_matrix.Fvu.transpose() * riccati_next.sv();
46}

58inline void BackwardRiccatiRecursionFactorizer::factorizeRiccatiFactorization(
59 const SplitRiccatiFactorization& riccati_next, SplitKKTMatrix& kkt_matrix,
60 const SplitKKTResidual& kkt_residual, const LQRPolicy& lqr_policy,
61 SplitRiccatiFactorization& riccati) {
62 GK_.noalias() = kkt_matrix.Quu * lqr_policy.K;
63 kkt_matrix.Qxx.noalias() -= lqr_policy.K.transpose() * GK_;
64 // Riccati factorization matrix with preserving the symmetry
65 riccati.P = 0.5 * (kkt_matrix.Qxx + kkt_matrix.Qxx.transpose());
66 // Riccati factorization vector
67 riccati.s.noalias() = kkt_matrix.Fxx.transpose() * riccati_next.s;
68 riccati.s.noalias() -= AtP_ * kkt_residual.Fx;
69 riccati.s.noalias() -= kkt_residual.lx;
70 riccati.s.noalias() -= kkt_matrix.Qxu * lqr_policy.k;
71}
