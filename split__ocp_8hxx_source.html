63inline void SplitOCP::evalOCP(Robot& robot, const ContactStatus& contact_status,
64 const double t, const double dt,
65 const SplitSolution& s,
66 const Eigen::VectorXd& q_next,
67 const Eigen::VectorXd& v_next,
68 SplitKKTResidual& kkt_residual) {
69 assert(dt > 0);
70 assert(q_next.size() == robot.dimq());
71 assert(v_next.size() == robot.dimv());
72 robot.updateKinematics(s.q, s.v, s.a);
73 kkt_residual.setContactStatus(contact_status);
74 kkt_residual.setZero();
75 stage_cost_ = cost_->evalStageCost(robot, cost_data_, t, dt, s);
76 constraints_->evalConstraint(robot, constraints_data_, s);
77 stage_cost_ += dt * constraints_data_.logBarrier();
78 state_equation_.evalStateEquation(robot, dt, s, q_next, v_next, kkt_residual);
79 contact_dynamics_.evalContactDynamics(robot, contact_status, s);
80}

296inline double SplitOCP::KKTError(const SplitKKTResidual& kkt_residual,
297 const double dt) const {
298 assert(dt > 0);
299 double err = 0;
300 err += kkt_residual.KKTError();
301 err += (dt*dt) * contact_dynamics_.KKTError();
302 err += (dt*dt) * constraints_data_.KKTError();
303 return err;
304}
