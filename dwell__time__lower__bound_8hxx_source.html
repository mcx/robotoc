15inline DwellTimeLowerBound::DwellTimeLowerBound(
16 const double barrier, const double fraction_to_boundary_rule)
17 : barrier_(barrier),
18 fraction_to_boundary_rule_(fraction_to_boundary_rule),
19 slack_(std::sqrt(barrier)),
20 dual_(std::sqrt(barrier)),
21 residual_(0),
22 cmpl_(0),
23 dslack_(0),

64inline void DwellTimeLowerBound::setSlack(const double min_dt, const double ts1,
65 const double ts2) {
66 assert(min_dt >= 0.0);
67 slack_ = ts2 - ts1 - min_dt;
68 if (slack_ <= 0) {
69 slack_ = std::sqrt(barrier_);
70 }
71 dual_ = barrier_ / slack_;
72}

75inline void DwellTimeLowerBound::evalConstraint(const double min_dt,
76 const double ts1,
77 const double ts2) {
78 assert(min_dt >= 0.0);
79 residual_ = ts1 + min_dt - ts2 + slack_;
80 cmpl_ = pdipm::computeComplementarySlackness(barrier_, slack_, dual_);
81 log_barrier_ = - barrier_ * std::log(slack_);
82}
