11inline SplitUnconstrOCP::SplitUnconstrOCP(
12 const Robot& robot, const std::shared_ptr<CostFunction>& cost,
13 const std::shared_ptr<Constraints>& constraints)
14 : cost_(cost),
15 cost_data_(cost->createCostFunctionData(robot)),
16 constraints_(constraints),
17 constraints_data_(constraints->createConstraintsData(robot, 0)),
18 unconstr_dynamics_(robot),
19 use_kinematics_(false),
20 stage_cost_(0) {
21 if (cost_->useKinematics() || constraints_->useKinematics()) {
22 use_kinematics_ = true;
23 }
24 try {
25 if (robot.hasFloatingBase()) {
26 throw std::logic_error(
27 "robot has floating base: robot should have no constraints!");
28 }
29 if (robot.maxPointContacts() > 0) {
30 throw std::logic_error(
31 "robot can have contacts: robot should have no constraints!");
32 }
33 }
34 catch(const std::exception& e) {
35 std::cerr << e.what() << '\n';
36 std::exit(EXIT_FAILURE);
37 }
38}

69inline void SplitUnconstrOCP::evalOCP(Robot& robot, const double t,
70 const double dt, const SplitSolution& s,
71 const Eigen::VectorXd& q_next,
72 const Eigen::VectorXd& v_next,
73 SplitKKTResidual& kkt_residual) {
74 assert(dt > 0);
75 assert(q_next.size() == robot.dimq());
76 assert(v_next.size() == robot.dimv());
77 if (use_kinematics_) {
78 robot.updateKinematics(s.q);
79 }
80 kkt_residual.setZero();
81 stage_cost_ = cost_->evalStageCost(robot, cost_data_, t, dt, s);
82 constraints_->evalConstraint(robot, constraints_data_, s);
83 stage_cost_ += dt * constraints_data_.logBarrier();
84 unconstr::stateequation::computeForwardEulerResidual(dt, s, q_next, v_next,
85 kkt_residual);
86 unconstr_dynamics_.evalUnconstrDynamics(robot, s);
87}

90inline void SplitUnconstrOCP::computeKKTResidual(Robot& robot, const double t,
91 const double dt,
92 const SplitSolution& s,
93 const SplitSolution& s_next,
94 SplitKKTMatrix& kkt_matrix,
95 SplitKKTResidual& kkt_residual) {
96 assert(dt > 0);
97 if (use_kinematics_) {
98 robot.updateKinematics(s.q);
99 }
100 kkt_residual.setZero();
101 stage_cost_ = cost_->linearizeStageCost(robot, cost_data_, t, dt, s,
102 kkt_residual);
103 constraints_->linearizeConstraints(robot, constraints_data_, dt, s,
104 kkt_residual);
105 stage_cost_ += dt * constraints_data_.logBarrier();
106 unconstr::stateequation::linearizeForwardEuler(dt, s, s_next,
107 kkt_matrix, kkt_residual);
108 unconstr_dynamics_.linearizeUnconstrDynamics(robot, dt, s, kkt_residual);
109 kkt_residual.kkt_error = KKTError(kkt_residual, dt);
110}

113inline void SplitUnconstrOCP::computeKKTSystem(Robot& robot, const double t,
114 const double dt,
115 const SplitSolution& s,
116 const SplitSolution& s_next,
117 SplitKKTMatrix& kkt_matrix,
118 SplitKKTResidual& kkt_residual) {
119 assert(dt > 0);
120 if (use_kinematics_) {
121 robot.updateKinematics(s.q);
122 }
123 kkt_matrix.setZero();
124 kkt_residual.setZero();
125 stage_cost_ = cost_->quadratizeStageCost(robot, cost_data_, t, dt, s,
126 kkt_residual, kkt_matrix);
127 constraints_->linearizeConstraints(robot, constraints_data_, dt, s,
128 kkt_residual);
129 stage_cost_ += dt * constraints_data_.logBarrier();
130 unconstr::stateequation::linearizeForwardEuler(dt, s, s_next,
131 kkt_matrix, kkt_residual);
132 unconstr_dynamics_.linearizeUnconstrDynamics(robot, dt, s, kkt_residual);
133 kkt_residual.kkt_error = KKTError(kkt_residual, dt);
134 constraints_->condenseSlackAndDual(constraints_data_, dt, s,
135 kkt_matrix, kkt_residual);
136 unconstr_dynamics_.condenseUnconstrDynamics(kkt_matrix, kkt_residual);
137}

178inline double SplitUnconstrOCP::KKTError(const SplitKKTResidual& kkt_residual,
179 const double dt) const {
180 assert(dt > 0);
181 double err = 0;
182 err += kkt_residual.KKTError();
183 err += (dt*dt) * unconstr_dynamics_.KKTError();
184 err += (dt*dt) * constraints_data_.KKTError();
185 return err;
186}
