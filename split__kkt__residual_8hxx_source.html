143inline bool SplitKKTResidual::isDimensionConsistent() const {
144 if (Fx.size() != 2*dimv_) return false;
145 if (lx.size() != 2*dimv_) return false;
146 if (la.size() != dimv_) return false;
147 if (lu.size() != dimu_) return false;
148 return true;

152inline bool SplitKKTResidual::isApprox(const SplitKKTResidual& other) const {
153 assert(isDimensionConsistent());
154 assert(other.isDimensionConsistent());
155 if (!Fx.isApprox(other.Fx)) return false;
156 if (!lx.isApprox(other.lx)) return false;
157 if (!la.isApprox(other.la)) return false;
158 if (!lu.isApprox(other.lu)) return false;
159 if (dimf_ > 0) {
160 if (!lf().isApprox(other.lf())) return false;
161 }
162 Eigen::VectorXd vec(4), other_vec(4);
163 vec << h, kkt_error, cost, constraint_violation;
164 other_vec << other.h, other.kkt_error, other.cost, other.constraint_violation;
165 if (!vec.isApprox(other_vec)) return false;
166 return true;
167}
